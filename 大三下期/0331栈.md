```c
typedef struct stack
{
    char data[MAX]; // 顺序表
    int top;        // 栈顶指针
} Stack;


void init(Stack *s)
{
    s->top = -1;
}
void push(Stack *s, char data)
{
    if (s->top == MAX - 1)
        return;
    s->data[++(s->top)] = data;
}
char pop(Stack *s)
{
    if (s->top == -1)
        return;
    return s->data[s->top--];
}
char top(Stack *s)
{
    return s->data[s->top];
}
void clean(Stack *s)
{
    s->top = -1;
}
bool empty(Stack *s)
{
    return s->top == -1;
}
```
