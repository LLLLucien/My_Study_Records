## 双向链表

```c
    //定义节点
    struct node
    {
        char data;
        struct node *next;
    };

    typedef struct node Node;
    //定义链表
    struct List
    {
        Node *head; //头节点
        Node *tail; //尾指针，空表时指向头节点
    };

    typedef struct List List;

    Node *add(List *l, ElemType ch)
    {
        // 创建新节点
        Node *node = malloc(sizeof(Node));
        // 节点传入数据
        node->data = ch;
        node->next = NULL;
        // 让当前的最后一个元素的后继，变为新节点

        l->tail->next = node;
        // 让尾指针指向新节点
        l->tail = node;
    }
```
